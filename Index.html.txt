<script>
const ETFs = ["CSUS.L","VWCG.L","IGLN.L","IBTA.L","EIMI.L"];
let prices = {};
let total = localStorage.getItem("total") || 0;

document.getElementById("total").innerText = total;

const etfSelect = document.getElementById("etf");
const switchSelect = document.getElementById("switchTo");

ETFs.forEach(e => {
  etfSelect.add(new Option(e, e));
  switchSelect.add(new Option(e, e));
});

function addMonthly() {
  let val = parseFloat(document.getElementById("monthly").value);
  if (!val) return;
  total = parseFloat(total) + val;
  localStorage.setItem("total", total);
  document.getElementById("total").innerText = total;
}

async function fetchPrices() {
  const url = "https://query1.finance.yahoo.com/v7/finance/quote?symbols=" + ETFs.join(",");
  const res = await fetch(url);
  const data = await res.json();

  data.quoteResponse.result.forEach(e => {
    prices[e.symbol] = e.regularMarketPrice;
  });

  document.getElementById("status").innerText = "✅ 已更新";
}

function calcBuy() {
  const etf = document.getElementById("etf").value;
  const amount = parseFloat(document.getElementById("amount").value);
  const price = prices[etf];

  if (!price || !amount) return;

  const qty = Math.floor(amount / price);
  const remain = amount - qty * price;

  document.getElementById("buyResult").innerHTML =
    `買 ${qty} 股<br>剩餘 ${remain.toFixed(2)}`;
}

function calcSwitch() {
  const val = parseFloat(document.getElementById("currentValue").value);
  const etf = document.getElementById("switchTo").value;
  const price = prices[etf];

  if (!val || !price) return;

  const qty = Math.floor(val / price);
  const remain = val - qty * price;

  document.getElementById("switchResult").innerHTML =
    `買 ${qty} 股<br>剩餘 ${remain.toFixed(2)}`;
}
</script>
